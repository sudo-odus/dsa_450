int isPlaindrome(string S)
	{
	    // Your code goes here
	    int size=S.length();
	    int i=0;
	    int res=1;
	    while(i<size/2)
	    {
	        if(S[i]!=S[size-i-1])
	        {
	            res=0;
	        }
	        i++;
	    }
	    return(res);
	}
